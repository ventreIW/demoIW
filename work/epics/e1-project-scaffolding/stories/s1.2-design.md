# s1.2 Design — Frontend Skeleton

---

## App Router structure

```
src/
  app/
    layout.tsx          ← RootLayout: sets lang="es", loads globals.css
    page.tsx            ← renders <MainLayout> with <PlaceholderContent>
    healthcheck/
      route.ts          ← NextResponse.json({ status: "ok" })
  components/
    layout/
      MainLayout.tsx    ← flex container: <Sidebar> + <main>
      Sidebar.tsx       ← fixed sidebar with nav links
    ui/                 ← shadcn/ui generated (do not edit manually)
  styles/
    globals.css         ← @tailwind base/components/utilities
  lib/
    utils.ts            ← cn() helper (shadcn/ui standard)
```

## Healthcheck route

```typescript
// src/app/healthcheck/route.ts
import { NextResponse } from "next/server"
export function GET() {
  return NextResponse.json({ status: "ok" })
}
```

Test:
```typescript
// src/app/healthcheck/route.test.ts
import { GET } from "./route"
it("returns 200 with status ok", async () => {
  const res = await GET()
  expect(res.status).toBe(200)
  expect(await res.json()).toEqual({ status: "ok" })
})
```
