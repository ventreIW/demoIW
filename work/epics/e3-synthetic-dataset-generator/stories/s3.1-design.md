## ProceduralGenerator structure

```python
class ProceduralGenerator:
    def __init__(self, params: GenerationParams):
        self._params = params
        self._rng = np.random.default_rng(params.seed)  # determinism anchor
        self._faker = Faker("es_MX")
        Faker.seed(params.seed)                          # Faker determinism

    def generate(self) -> RawDataset:
        clients = self._generate_clients()
        invoices = self._generate_invoices(clients)
        payments = self._generate_payments(invoices)
        return RawDataset(clients=clients, invoices=invoices, payments=payments)
```

## Client generation

```python
def _generate_clients(self) -> pd.DataFrame:
    n = self._params.client_count
    patterns = self._sector_payment_patterns()  # sector-weighted choices
    return pd.DataFrame({
        "id": [str(uuid4()) for _ in range(n)],
        "name": [self._faker.company() for _ in range(n)],  # later replaced by LLM
        "sector": self._params.sector.value,
        "payment_history_pattern": self._rng.choice(
            list(PaymentPattern), size=n, p=patterns
        ),
    })
```

## Invoice generation

```python
def _generate_invoices(self, clients: pd.DataFrame) -> pd.DataFrame:
    rows = []
    for _, client in clients.iterrows():
        n_invoices = max(1, int(self._rng.poisson(self._params.invoice_volume)))
        for _ in range(n_invoices):
            amount = max(100, self._rng.normal(
                float(self._params.amount_mean),
                float(self._params.amount_std)
            ))
            due_date = date.today() - timedelta(
                days=int(self._rng.exponential(self._params.days_overdue_lambda))
            )
            is_overdue = self._rng.random() < self._params.overdue_rate
            rows.append({
                "id": str(uuid4()),
                "client_id": client["id"],
                "amount": round(Decimal(str(amount)), 2),
                "due_date": due_date,
                "status": "overdue" if is_overdue else "paid",
            })
    return pd.DataFrame(rows)
```
