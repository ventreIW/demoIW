## ORM model structure (SQLAlchemy 2.0)

```python
# app/adapters/persistence/models.py
class Base(DeclarativeBase): pass

class ScenarioORM(Base):
    __tablename__ = "scenarios"
    id: Mapped[UUID] = mapped_column(primary_key=True, default=uuid4)
    name: Mapped[str]
    sector: Mapped[str]          # stored as string, domain layer converts to enum
    seed: Mapped[int | None]
    parameters: Mapped[dict] = mapped_column(JSONB)
    source: Mapped[str]
    status: Mapped[str]
    created_at: Mapped[datetime] = mapped_column(server_default=func.now())

class ClientORM(Base):
    __tablename__ = "clients"
    id: Mapped[UUID] = mapped_column(primary_key=True, default=uuid4)
    scenario_id: Mapped[UUID] = mapped_column(ForeignKey("scenarios.id", ondelete="CASCADE"))
    name: Mapped[str]
    sector_description: Mapped[str | None]
    payment_history_pattern: Mapped[str]
    # total_outstanding computed at query time via invoice join
```

Same pattern for `InvoiceORM`, `PaymentORM`, `ScoreORM`, `CommunicationORM`, `ContactResultORM`.

## Mapper pattern

```python
# app/adapters/persistence/mappers.py
def scenario_orm_to_domain(orm: ScenarioORM) -> Scenario:
    return Scenario(
        id=orm.id,
        name=orm.name,
        sector=Sector(orm.sector),
        ...
    )

def scenario_domain_to_orm(domain: Scenario) -> ScenarioORM:
    return ScenarioORM(
        id=domain.id,
        name=domain.name,
        sector=domain.sector.value,
        ...
    )
```
